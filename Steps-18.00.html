Adding all KPIs to the dash board 18


Repeat the steps for all the KPIs.
KPI-Fatal Rate%




.Fatal =
 COALESCE( CALCULATE('TFL Bus Safety v.2'[Current Year Incidents], 'TFL Bus Safety v.2'[Injury Result Description]= "Fatal"),0)
.LY Fatal = 
VAR SelectedYear =MAX('CalendarTable'[Year])
RETURN 
  COALESCE(CALCULATE('TFL Bus Safety v.2'[.Fatal],
        'CalendarTable'[Year] = SelectedYear - 1
    ),0) 
Fatal Rate% = 
 COALESCE(DIVIDE('TFL Bus Safety v.2'[.Fatal],'TFL Bus Safety v.2'[Current Year Incidents]),0)
.Fatal YoY% =
 IF ([.LY Fatal] > 0, (DIVIDE('TFL Bus Safety v.2'[.Fatal]-'TFL Bus Safety v.2'[.LY Fatal],'TFL Bus Safety v.2'[.LY Fatal])), IF ([.Fatal] > 0, 1, "N/A"))
.Arrow Fatal Yoy% =
 IF (OR ([.Fatal YoY%] = "N/A", [.Fatal YoY%] =0), UNICHAR(10134) , SWITCH(TRUE, [.Fatal YoY%] < 0,UNICHAR(9660),  UNICHAR(9650) ))

KPI - Injury Rate%



.Injury Count = 
CALCULATE('TFL Bus Safety v.2'[Current Year Incidents]-'TFL Bus Safety v.2'[.Fatal])
.LY Injury Count = 
VAR SelectedYear =MAX('CalendarTable'[Year])
RETURN 
  COALESCE(CALCULATE('TFL Bus Safety v.2'[.Injury Count],
        'CalendarTable'[Year] = SelectedYear - 1
    ),0) 
Injury Rate% = 
VAR TotalInjuries = 'TFL Bus Safety v.2'[Current Year Incidents]-'TFL Bus Safety v.2'[.Fatal]
VAR TotalIncidents ='TFL Bus Safety v.2'[Current Year Incidents]
RETURN 
    IF(TotalIncidents = TotalInjuries, 1, (TotalInjuries / TotalIncidents))
.Injury YoY% =
 COALESCE(DIVIDE(([.Injury Count]-[.LY Injury Count]),[.LY Injury Count]),0)
.Arrow Injury Yoy% = 
SWITCH(TRUE(),[.Injury YoY%] < 0,UNICHAR(9660),UNICHAR(9650))
KPI – Hospitalization Rate%



.Treated_on_Scene = 
COALESCE( CALCULATE('TFL Bus Safety v.2'[Current Year Incidents], 'TFL Bus Safety v.2'[Injury Result Description] = "Injuries treated on scene"),0)
.Minor_Injuries = 
COALESCE( CALCULATE('TFL Bus Safety v.2'[Current Year Incidents],'TFL Bus Safety v.2'[Injury Result Description] = "Reported Minor Injury - Treated at Hospital"),0)
.Serious_Injuries =
 COALESCE( CALCULATE('TFL Bus Safety v.2'[Current Year Incidents],'TFL Bus Safety v.2'[Injury Result Description] = "Taken to Hospital – Reported Serious Injury or Severity Unknown"),0)
.Hospitalization Count =
 CALCULATE('TFL Bus Safety v.2'[Current Year Incidents]-('TFL Bus Safety v.2'[.Fatal] + 'TFL Bus Safety v.2'[.Treated_on_Scene]+'TFL Bus Safety v.2'[.Minor_Injuries]) )
.LY Hospitalization Count =
 VAR 
SelectedYear =MAX('CalendarTable'[Year])
RETURN 
  COALESCE(CALCULATE('TFL Bus Safety v.2'[.Hospitalization Count],
        'CalendarTable'[Year] = SelectedYear - 1
    ),0) 
Hospitalization Rate% = 
DIVIDE( 'TFL Bus Safety v.2'[.Hospitalization Count], [Current Year Incidents], 0 ) 
.Hospitalization YoY% = 
COALESCE(DIVIDE('TFL Bus Safety v.2'[.Hospitalization Count]-'TFL Bus Safety v.2'[.LY Hospitalization Count],'TFL Bus Safety v.2'[.LY Hospitalization Count]),0) 
.Arrow Hospitalization Yoy% =
 SWITCH(TRUE(),[.Hospitalization YoY%] < 0,UNICHAR(9660),UNICHAR(9650))
